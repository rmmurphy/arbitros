87  /*------------------------------------------------------------------------*
88  * This function assumes that it has been called within the context of
89  * mutual exclusion. Therefore, it makes sure that the SREG is saved with
90  * the GIE bit set. Interrupts will not get turned back on until a thread
91  * with the GIE bit enabled is restored with the call to
92  * 'arb_contextRestore'.
93  *------------------------------------------------------------------------*/
94  static inline void __attribute__((always_inline)) arb_contextSaveWithIntsOn\
95  ( void)
96  {
97 
98  /*---------------------------------------------------------------------*
99  * !Note! The XMEGA pushes the program counter onto the stack when a
100  * thread gets preempted or makes a subroutine call. Upon entering this
101  * this routine the stack pointer is pointing to the thread's
102  * program counter.
103  *---------------------------------------------------------------------*/
104 
105  /*---------------------------------------------------------------------*
106  * Push the rest of the XMEGA registers SREG, and r0->r31 onto the
107  * thread's stack (starting with the first address after the PC).
108  * After storing R1 we need to set it to zero so we are performing the
109  * same operation as the compiler if it were explicitly doing the
110  * context save for us.
111  *---------------------------------------------------------------------*/
112  asm(" push r0");
113 
114  /*---------------------------------------------------------------------*
115  * We disabled interrupts before calling this function, therefore we
116  * need to make sure that the SREG is saved with the GIE bit set.
117  *---------------------------------------------------------------------*/
118  asm( "in r0, __SREG__");
119  asm(" push r16"); /*Save the contents of r16 so we can use it*/
120  asm( "mov r16, r0"); /*Can't directly set a bit in r0 so copy it to
121  r16*/
122  asm( "sbr r16, 128"); /*Turn interrupts back on in the stored SREG*/
123  asm( "mov r0, r16"); /*Copy the change back into r0*/
124  asm( "pop r16"); /*Get back the old value of r16*/
125  asm( "push r0"); /*Save SREG with GIE enabled*/
126  asm( "push r1");
127  asm( "clr r1");

244  static inline void __attribute__((always_inline)) arb_contextRestore( void)
245  {
246 
247  /*---------------------------------------------------------------------*
248  * Get the current thread's stack pointer from the TCB and load the
249  * CPU stack register
250  *---------------------------------------------------------------------*/
251  pt_csTemp = (t_tcb *)UTL_GET_LINK_ELEMENT_PTR_CONT_CURR(gt_activeThreads);
252  SPL = (uint8_t)((pt_csTemp->s_sP) & 0x00FF);
253  SPH = (uint8_t)(((pt_csTemp->s_sP) >> 8) & 0x00FF);
254 
255  /*---------------------------------------------------------------------*
256  * Now that the CPU stack ptr is pointing to the stack for this thread,
257  * restore the general purpose registers to the place where the thread
258  * last left off by popping there values off the thread's stack.
259  *---------------------------------------------------------------------*/
260  asm( "pop r31");
261  asm( "pop r30");
262  asm( "pop r29");
263  asm( "pop r28");
264  asm( "pop r27");
265  asm( "pop r26");
