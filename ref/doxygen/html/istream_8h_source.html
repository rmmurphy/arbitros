291  if ((T)-1 < 0) {

294  // max absolute value of negative number is m + 1.
295  if (getNumber(m, m + 1, &tmp)) {
296  *value = (T)tmp;
297  }
298  } else {
299  // max unsigned value for T
300  uint32_t const m = (T)-1;
